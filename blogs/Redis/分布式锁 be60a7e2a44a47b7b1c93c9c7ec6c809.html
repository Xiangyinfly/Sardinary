    分布式锁

    Sardinary Chen 2/26/2024 Redis微服务

    # 分布式锁要满足的特性

    # 独占性

    任何时候只能有且仅有一个线程持有

    # 高可用

    不会因为某个节点挂掉导致获取锁和释放锁失败问题

    高并发下的可用性

    # 防死锁

    超时机制或撤销操作

    # 不乱抢

    不能unlock别人的锁

    自己加的锁必须自己释放

    # 重入性

    同一个节点同一个线程获取锁之后，还可以再次获得这个锁

    # 一个相对完善的方案

    # 技术要点

    设置过期时间防止死锁

    未抢到锁的时候while重试自旋

    设置自动续期防止业务未完成但是锁过期的情况

    使用lua脚本保证redis操作原子性

    使用hset实现锁的可重入性

    判定锁的归属防止误删别人的锁

    使用工厂设计模式封装

    # redis分布式锁的实现

    package com.xiang;

import lombok.SneakyThrows;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.data.redis.core.script.DefaultRedisScript;

import java.util.List;
import java.util.Timer;
import java.util.TimerTask;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.locks.Condition;
import java.util.concurrent.locks.Lock;

public class RedisDistributedLock implements Lock {

    private StringRedisTemplate redisTemplate;
    private String lockName;
    private String value;
    private Long expireTime;

    public RedisDistributedLock(StringRedisTemplate redisTemplate, String lockName, String uuid) {
        this.redisTemplate = redisTemplate;
        this.lockName = lockName;
        this.value = uuid + ":" + Thread.currentThread().getId();
        this.expireTime = 50L;
    }

    @Override
    public void lock() {
        tryLock();
    }

    @Override
    @SneakyThrows
    public boolean tryLock() {
        tryLock(-1L, TimeUnit.SECONDS);
        return false;
    }

    @Override
    public boolean tryLock(long time, TimeUnit unit) throws InterruptedException {
        if (time == -1L) {
            String script = "if redis.call('exists',KEYS[1]) == 0 or redis.call('hexists',KEYS[1],ARGV[1]) == 1 then " +
                    "redis.call('hincrby',KEYS[1],ARGV[1],1) " +
                    "redis.call('expire',KEYS[1],ARGV[2]) " +
                    "return 1 " +
                    "else " +
                    "return 0 " +
                    "end";
						//1. 这里如果拿不到分布式锁，就自旋直到拿到。不用方法递归，容易栈溢出
		        //2. 设置锁过期时间，防止机器挂了出现死锁
		        //3. 加锁和设置过期时间要保证原子性，防止某些原因加完锁未加过期时间导致死锁
            while (Boolean.FALSE.equals(redisTemplate.execute(new DefaultRedisScript<>(script, Boolean.class), List.of(lockName), value, String.valueOf(expireTime)))) {
                TimeUnit.MILLISECONDS.sleep(60);
            }
            //加锁成功后调用自动续期程序
            //TODO 如果业务程逻辑一直阻塞或者挂了怎么办？一直续期导致死锁
            renewExpireTime();
        }
        return false;
    }

    private void renewExpireTime() {
        String script = "if redis.call('HEXISTS',KEYS[1],ARGV[1]) == 1 then " +
                "return redis.call('expire',KEYS[1],ARGV[2]) " +
                "else" +
                "return 0 " +
                "end";
        new Timer().schedule(new TimerTask() {
            @Override
            public void run() {
                if (Boolean.TRUE.equals(redisTemplate.execute(new DefaultRedisScript<>(script, Boolean.class), List.of(lockName), value, String.valueOf(expireTime)))) {
                    //续期成功则继续监视续期
                    renewExpireTime();
                }
            }
        }, (this.expireTime * 1000) / 3);
    }

    @Override
    public void unlock() {
        String script = "if redis.call('HEXISTS',KEYS[1],ARGV[1]) == 0 then  " +
                "return nil  " +
                "elseif redis.call('HINCRBY',KEYS[1],ARGV[1],-1) == 0 then  " +
                "return redis.call('del',KEYS[1])  " +
                "else  " +
                "return 0 " +
                "end ";
        Long flag = redisTemplate.execute(new DefaultRedisScript<>(script, Long.class), List.of(lockName), value);
        if (null == flag) {
            throw new RuntimeException("lock does not exist!");
        }
    }

    @Override
    public void lockInterruptibly() throws InterruptedException {

    }

    @Override
    public Condition newCondition() {
        return null;
    }
}

    # 工厂设计模式实现

    package com.xiang;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.redis.core.StringRedisTemplate;
import org.springframework.stereotype.Component;

import java.util.UUID;
import java.util.concurrent.locks.Lock;

@Component
public class DistributedLockFactory {
    @Autowired
    private StringRedisTemplate redisTemplate;
    private String lockName;
    //此uuid可以看作当前服务器的唯一标识
    //将uuid在工厂类就确定，保证new两次锁的时候的可重入性
    private String uuid;

    public DistributedLockFactory() {
        this.uuid = UUID.randomUUID().toString();
    }

    public Lock getDistributedLock(String lockType) {
        if (lockType == null) {
            return null;
        }
        if (lockType.equalsIgnoreCase("REDIS")) {
            lockName = "redisLock";
            return new RedisDistributedLock(redisTemplate,lockName,uuid);
        }
        //......
        //TODO 增加其他的分布式锁实现方式

        return null;
    }
}
